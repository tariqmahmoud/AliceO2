31  void process(soa::Join<aod::BCs, aod::Run2BCInfos, aod::BcSels>::iterator const& bc)
32  {
33  // Fill fired trigger classes
34  uint64_t triggerMask = bc.triggerMask();
35  uint64_t triggerMaskNext50 = bc.triggerMaskNext50();
36  for (int i = 0; i < 50; i++) {
37  if (triggerMask & 1ull << i) {
38  hFiredClasses->Fill(i);
39  }
40  if (triggerMaskNext50 & 1ull << i) {
41  hFiredClasses->Fill(i + 50);
42  }
43  }
44 
45  // Fill fired aliases
46  for (int i = 0; i < kNaliases; i++) {
47  if (bc.alias()[i]) {
48  hFiredAliases->Fill(i);
49  }
50  }
51  }
52 };

75  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Run2MatchedSparse>::iterator const& col,
76  aod::Zdcs const& zdcs,
77  aod::FV0As const& fv0as,
78  aod::FV0Cs const& fv0cs,
79  aod::FT0s ft0s,

85 
86  float timeZNA = col.has_zdc() ? col.zdc().timeZNA() : -999.f;

88  float timeV0A = col.has_fv0a() ? col.fv0a().time() : -999.f;
89  float timeV0C = col.has_fv0c() ? col.fv0c().time() : -999.f;
90  float timeT0A = col.has_ft0() ? col.ft0().timeA() : -999.f;
91  float timeT0C = col.has_ft0() ? col.ft0().timeC() : -999.f;
92  float timeFDA = col.has_fdd() ? col.fdd().timeA() : -999.f;
93  float timeFDC = col.has_fdd() ? col.fdd().timeC() : -999.f;
94 
95  hTimeV0Aall->Fill(timeV0A);
96  hTimeV0Call->Fill(timeV0C);
97  hTimeZNAall->Fill(timeZNA);
98  hTimeZNCall->Fill(timeZNC);
99  hTimeT0Aall->Fill(timeT0A);
100  hTimeT0Call->Fill(timeT0C);
101  hTimeFDAall->Fill(timeFDA);
102  hTimeFDCall->Fill(timeFDC);
103 
104  if (selection == 7 && !col.sel7()) {
105  return;
106  }
107 
108  if (selection == 8 && !col.sel8()) {
109  return;
110  }
111 
112  if (selection != 7 && selection != 8) {
113  LOGF(fatal, "Unknown selection type! Use `--sel 7` or `--sel 8`");
114  }
115 
116  hTimeV0Aacc->Fill(timeV0A);
117  hTimeV0Cacc->Fill(timeV0C);

119  hTimeZNCacc->Fill(timeZNC);

122  hTimeFDAacc->Fill(timeFDA);
123  hTimeFDCacc->Fill(timeFDC);
124  }
