62  template <typename T1, typename T2>
63  bool selectionTopol(const T1& hfCandProng2, const T2& trackPositron, const T2& trackElectron)
64  {
65  auto candpT = hfCandProng2.pt();
66  auto pTBin = findBin(pTBins, candpT);
67  if (pTBin == -1) {
68  return false;
69  }
70 
71  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
72  return false; //check that the candidate pT is within the analysis range
73  }
74 
75  if (TMath::Abs(InvMassJpsiToEE(hfCandProng2) - RecoDecay::getMassPDG(pdg::code::kJpsi)) > cuts->get(pTBin, "m")) {
76  return false;
77  }
78 
79  if (trackElectron.pt() < cuts->get(pTBin, "pT El") || trackPositron.pt() < cuts->get(pTBin, "pT El")) {
80  return false; //cut on daughter pT
81  }
82  if (TMath::Abs(trackElectron.dcaPrim0()) > cuts->get(pTBin, "DCA_xy") || TMath::Abs(trackPositron.dcaPrim0()) > cuts->get(pTBin, "DCA_xy")) {
83  return false; //cut on daughter dca - need to add secondary vertex constraint here
84  }
85  if (TMath::Abs(trackElectron.dcaPrim1()) > cuts->get(pTBin, "DCA_z") || TMath::Abs(trackPositron.dcaPrim1()) > cuts->get(pTBin, "DCA_z")) {
86  return false; //cut on daughter dca - need to add secondary vertex constraint here
87  }
88 
89  return true;
90  }

137  {
138 
139  for (auto& hfCandProng2 : hfCandProng2s) { //looping over 2 prong candidates
140 
141  auto trackPos = hfCandProng2.index0_as<aod::BigTracksPID>(); //positive daughter
142  auto trackNeg = hfCandProng2.index1_as<aod::BigTracksPID>(); //negative daughter
143 
144  if (!(hfCandProng2.hfflag() & 1 << JpsiToEE)) {
145  hfSelJpsiToEECandidate(0);
146  continue;
147  }
148 
149  // daughter track validity selection
150  if (!daughterSelection(trackPos) || !daughterSelection(trackNeg)) {
151  hfSelJpsiToEECandidate(0);
152  continue;
153  }
154 
155  //implement filter bit 4 cut - should be done before this task at the track selection level
156  //need to add special cuts (additional cuts on decay length and d0 norm)
157 
158  if (!selectionTopol(hfCandProng2, trackPos, trackNeg)) {
159  hfSelJpsiToEECandidate(0);
160  continue;
161  }
162 
163  if (selectionPID(trackPos) == 0 || selectionPID(trackNeg) == 0) {
164  hfSelJpsiToEECandidate(0);
165  continue;
166  }

168  hfSelJpsiToEECandidate(1);
