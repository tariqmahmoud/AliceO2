68 static void patchPage(gsl::span<const std::byte> rdhBuffer, bool verbose)
69 {
70  static int sNrdhs = 0;
71  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
72  sNrdhs++;
73 
74  auto cruId = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF;
75  auto flags = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF00;
76  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhAny);
77  auto existingFeeId = o2::raw::RDHUtils::getFEEID(rdhAny);
78  if (existingFeeId == 0) {
79  // early versions of raw data did not set the feeId
80  // which we need to select the right decoder
81  uint32_t feeId = cruId * 2 + endpoint + flags;
82  o2::raw::RDHUtils::setFEEID(rdhAny, feeId);
83  }
84 
85  if (verbose) {
86  std::cout << "RDH number " << sNrdhs << "--\n";
87  o2::raw::RDHUtils::printRDH(rdhAny);
88  }
89 };

111 void DataDecoder::decodeBuffer(gsl::span<const std::byte> buf)
112 {
113  if (mDebug) {
114  std::cout << "\n\n============================\nStart of new buffer\n";
115  }
116  size_t bufSize = buf.size();
117  size_t pageStart = 0;
118  while (bufSize > pageStart) {
119  RDH* rdh = reinterpret_cast<RDH*>(const_cast<std::byte*>(&(buf[pageStart])));
120  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
121  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
122  if (rdhHeaderSize != 64) {
123  break;
124  }
125  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
126 
127  gsl::span<const std::byte> page(reinterpret_cast<const std::byte*>(rdh), pageSize);
128  decodePage(page);
129 
130  pageStart += pageSize;
131  }
132 }

134 void DataDecoder::dumpDigits(bool bending)
135 {
136  if (mOutputDigits.size() != mSampaInfos.size()) {
137  return;
138  }
139 
140  for (size_t di = 0; di < mOutputDigits.size(); di++) {
141  auto& d = mOutputDigits[di];
142  auto& t = mSampaInfos[di];
143  if (d.getPadID() < 0) {
144  continue;
145  }
146  const Segmentation& segment = segmentation(d.getDetID());
147  bool bend = segment.isBendingPad(d.getPadID());
148  if (bending != segment.isBendingPad(d.getPadID())) {

150  }
151  float X = segment.padPositionX(d.getPadID());
152  float Y = segment.padPositionY(d.getPadID());
153  uint32_t orbit = t.orbit;
154  uint32_t bunchCrossing = t.bunchCrossing;
155  uint32_t sampaTime = t.sampaTime;
156  std::cout << fmt::format(" DE {:4d} PAD {:5d} ADC {:6d} TIME ({} {} {:4d})",
157  d.getDetID(), d.getPadID(), d.getADC(), orbit, bunchCrossing, sampaTime);
158  std::cout << fmt::format("\tC {} PAD_XY {:+2.2f} , {:+2.2f}", (bending ? (int)0 : (int)1), X, Y);

171 void DataDecoder::decodePage(gsl::span<const std::byte> page)
172 {
173  if (mDebug) {
174  std::cout << "\n----------------------------\nStart of new page\n\n";
175  }
176  size_t ndigits{0};
177 
178  auto heartBeatHandler = [&](DsElecId dsElecId, uint8_t chip, uint32_t bunchCrossing) {
179  SampaInfo sampaId;
180  sampaId.chip = chip;
181  sampaId.ds = dsElecId.elinkId();
182  sampaId.solar = dsElecId.solarId();
183 
184  mSampaTimeFrameStarts[sampaId.id].emplace(mOrbit, bunchCrossing);
185  if (mDebug) {
186  std::cout << "HeartBeat: solar " << sampaId.solar << " ds " << sampaId.ds << " chip " << sampaId.chip
187  << " mOrbit " << mOrbit << " bunchCrossing " << bunchCrossing << std::endl;
188  }
189  };
190 
191  auto channelHandler = [&](DsElecId dsElecId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
192  if (mChannelHandler) {
193  mChannelHandler(dsElecId, channel, sc);
194  }
195 
196  if (mDs2manu) {
197  channel = ds2manu(int(channel));
198  }
199 
200  uint32_t digitadc = sc.sum();
201 
202  int deId{-1};
203  int dsIddet{-1};
204  if (auto opt = mElec2Det(dsElecId); opt.has_value()) {
205  DsDetId dsDetId = opt.value();
206  dsIddet = dsDetId.dsId();
207  deId = dsDetId.deId();
208  }
209  if (mDebug) {
210  auto s = asString(dsElecId);
211  auto ch = fmt::format("{}-CH{:02d}", s, channel);
212  std::cout << ch << " "
213  << "deId " << deId << " dsIddet " << dsIddet << std::endl;
214  }
215 
216  if (deId < 0 || dsIddet < 0 || !isValidDeID(deId)) {
217  return;
218  }
219 
220  int padId = -1;
221  const Segmentation& segment = segmentation(deId);
222 
223  padId = segment.findPadByFEE(dsIddet, int(channel));
224  if (mDebug) {
225  auto s = asString(dsElecId);
226  auto ch = fmt::format("{}-CH{:02d}", s, channel);
227  std::cout << ch << " "
228  << fmt::format("PAD ({:04d} {:04d} {:04d})\tADC {:06d} TIME ({} {} {:02d}) SIZE {} END {}",
229  deId, dsIddet, padId, digitadc, mOrbit, sc.bunchCrossing, sc.sampaTime, sc.nofSamples(), (sc.sampaTime + sc.nofSamples() - 1))
230  << (((sc.sampaTime + sc.nofSamples() - 1) >= 98) ? " *" : "") << std::endl;
231  }
232 
233  // skip channels not associated to any pad
234  if (padId < 0) {
235  return;
236  }
237 
238  SampaInfo sampaInfo;
239  sampaInfo.chip = channel / 32;
240  sampaInfo.ds = dsElecId.elinkId();
241  sampaInfo.solar = dsElecId.solarId();
242  sampaInfo.sampaTime = sc.sampaTime;
243  sampaInfo.bunchCrossing = sc.bunchCrossing;
244  sampaInfo.orbit = mOrbit;
245 
246  mOutputDigits.emplace_back(o2::mch::Digit(deId, padId, digitadc, 0, sc.nofSamples()));
247  mSampaInfos.emplace_back(sampaInfo);
248 
249  if (mDebug) {
250  std::cout << "DIGIT STORED:\nADC " << mOutputDigits.back().getADC() << " DE# " << mOutputDigits.back().getDetID() << " PadId " << mOutputDigits.back().getPadID() << " time " << mSampaInfos.back().sampaTime << std::endl;
251  }
252  ++ndigits;
253  };
254 
255  patchPage(page, mDebug);
256 
257  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(page[0])));
258  mOrbit = o2::raw::RDHUtils::getHeartBeatOrbit(rdhAny);
259  if (mDebug) {
260  std::cout << "[decodeBuffer] mOrbit set to " << mOrbit << std::endl;
261  }
262 
263  if (mRdhHandler) {
264  mRdhHandler(&rdhAny);
265  }
266 
267  // add orbit to vector if not present yet
268  mOrbits.emplace(page);
269 
270  if (!mDecoder) {
271  DecodedDataHandlers handlers;
272  handlers.sampaChannelHandler = channelHandler;
273  handlers.sampaHeartBeatHandler = heartBeatHandler;
274  mDecoder = mFee2Solar ? o2::mch::raw::createPageDecoder(page, handlers, mFee2Solar)
275  : o2::mch::raw::createPageDecoder(page, handlers);
276  }
277  mDecoder(page);
278 
279  if (mDebug) {
280  std::cout << "[decodeBuffer] mOrbits size: " << mOrbits.size() << std::endl;
281  //dumpOrbits(mOrbits);
282  std::cout << "[decodeBuffer] mOutputDigits size: " << mOutputDigits.size() << std::endl;
283  dumpDigits(true);
284  dumpDigits(false);
285  }

288 int32_t DataDecoder::digitsTimeDiff(uint32_t orbit1, uint32_t bc1, uint32_t orbit2, uint32_t bc2)
289 {
290  // bunch crossings are stored with 20 bits
291  static const int32_t BCROLLOVER = (1 << 20);
292  // bunch crossings half range
293  //static const int32_t BCHALFRANGE = (1 << 19);
294  // number of bunch crossings in one orbit
295  static const int32_t BCINORBIT = 3564;
296 
297  // We use the difference of orbits values to estimate the minimum and maximum allowed
298  // difference in bunch crossings
299  int64_t dOrbit = static_cast<int64_t>(orbit2) - static_cast<int64_t>(orbit1);
300 
301  // Digits might be sent out later than the orbit in which they were recorded.
302  // We account for this by allowing an extra +/- 2 orbits when converting the
303  // difference from orbit numbers to bunch crossings.
304  int64_t dBcMin = (dOrbit - 2) * BCINORBIT;
305  int64_t dBcMax = (dOrbit + 2) * BCINORBIT;
306 
307  // Difference in bunch crossing values
308  int64_t dBc = static_cast<int64_t>(bc2) - static_cast<int64_t>(bc1);
309 
310  if (dBc < dBcMin) {
311  // the difference is too small, so we assume that it needs to be
312  // incremented by one rollover factor
313  dBc += BCROLLOVER;
314  } else if (dBc > dBcMax) {
315  // the difference is too big, so we assume that it needs to be
316  // decremented by one rollover factor
317  dBc -= BCROLLOVER;
318  }
319 
320  return static_cast<int32_t>(dBc);
321 }

323 void DataDecoder::computeDigitsTime_(std::vector<o2::mch::Digit>& digits, std::vector<SampaInfo>& sampaInfo, SampaTimeFrameStarts& sampaTimeFrameStarts, bool debug)
324 {
325  if (digits.size() != sampaInfo.size()) {
326  throw std::logic_error(fmt::format("{} mismatch in digits and sampaInfo vector sizes ({} and {})\n",
327  __PRETTY_FUNCTION__, digits.size(), sampaInfo.size()));
328  }
329 
330  auto setDigitTime = [&](Digit& d, int32_t tfTime) {
331  d.setTime(tfTime);
332  if (debug) {
333  std::cout << "[computeDigitsTime_] hit time set to " << d.getTime() << std::endl;
334  }
335  };
336 
337  for (size_t di = 0; di < digits.size(); di++) {
338  Digit& d = digits[di];
339  SampaInfo& info = sampaInfo[di];
340  std::optional<SampaTimeFrameStart>& tfStart = sampaTimeFrameStarts[info.id];
341 
342  if (!tfStart) {
343  d.setTime(0xFFFFFFFF);
344  continue;
345  }
346 
347  int32_t tfTime = 0;
348  uint32_t bc = tfStart->mBunchCrossing;
349  uint32_t orbit = tfStart->mOrbit;
350  tfTime = DataDecoder::digitsTimeDiff(orbit, bc, info.orbit, info.getBXTime());
351  if (debug) {
352  std::cout << "[computeDigitsTime_] hit " << info.orbit << "," << info.getBXTime()
353  << " tfTime(1) " << orbit << "," << bc << " diff " << tfTime << std::endl;
354  }
355 
356  setDigitTime(d, tfTime);
357  }
358 }
