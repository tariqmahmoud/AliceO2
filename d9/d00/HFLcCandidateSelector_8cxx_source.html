116  return false; //cut on daughter pT

210  template <typename T>
211  int selectionPID(const T& track, int nPDG)
212  {
213  int statusTPC = -1;
214  int statusTOF = -1;
215 
216  if (validTPCPID(track)) {
217  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPC)) {
218  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPCCombined)) {
219  statusTPC = 0; //rejected by PID
220  } else {
221  statusTPC = 1; //potential to be acceepted if combined with TOF
222  }
223  } else {
224  statusTPC = 2; //positive PID
225  }
226  } else {
227  statusTPC = -1; //no PID info
228  }
229 
230  if (validTOFPID(track)) {
231  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOF)) {
232  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOFCombined)) {
233  statusTOF = 0; //rejected by PID
234  } else {
235  statusTOF = 1; //potential to be acceepted if combined with TOF
236  }
237  } else {
238  statusTOF = 2; //positive PID
239  }
240  } else {
241  statusTOF = -1; //no PID info
242  }
243 
244  if (statusTPC == 2 || statusTOF == 2) {
245  return 1; //what if we have 2 && 0 ?
246  } else if (statusTPC == 1 && statusTOF == 1) {
247  return 1;
248  } else if (statusTPC == 0 || statusTOF == 0) {
249  return 0;
250  } else {
251  return -1;

255  void process(aod::HfCandProng3 const& hfCandProng3s, aod::BigTracksPID const&)
256  {
257  for (auto& hfCandProng3 : hfCandProng3s) {
258  // final selection flag : 0-rejected 1-accepted
259  auto statusLcpKpi = 0;
260  auto statusLcpiKp = 0;
261 
262  if (!(hfCandProng3.hfflag() & 1 << LcToPKPi)) {
263  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
264  continue;
265  }
266  auto trackPos1 = hfCandProng3.index0_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
267  auto trackNeg1 = hfCandProng3.index1_as<aod::BigTracksPID>(); //negative daughter (positive for the antiparticles)
268  auto trackPos2 = hfCandProng3.index2_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
269 
270  // daughter track validity selection
271  if (!daughterSelection(trackPos1) || !daughterSelection(trackNeg1) || !daughterSelection(trackPos2)) {
272  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
273  continue;
274  }
275 
276  auto pidLcpKpi = -1;
277  auto pidLcpiKp = -1;
278 
279  //implement filter bit 4 cut - should be done before this task at the track selection level
280 
281  //conjugate independent topological selection
282  if (!selectionTopol(hfCandProng3)) {
283  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
284  continue;
285  }
286 
287  //conjugate dependent toplogical selection for Lc
288 
289  bool topolLcpKpi = selectionTopolConjugate(hfCandProng3, trackPos1, trackNeg1, trackPos2);
290  bool topolLcpiKp = selectionTopolConjugate(hfCandProng3, trackPos2, trackNeg1, trackPos1);
291 
292  if (!topolLcpKpi && !topolLcpiKp) {
293  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
294  continue;
295  }
296 
297  if (!d_FilterPID) {
298  // PID non applied
299  pidLcpKpi = 1;
300  pidLcpiKp = 1;
301  } else {
302  int proton1 = selectionPID(trackPos1, kProton);
303  int proton2 = selectionPID(trackPos2, kProton);
304  int kaonMinus = selectionPID(trackNeg1, kKPlus);
305  int pionPlus1 = selectionPID(trackPos1, kPiPlus);
306  int pionPlus2 = selectionPID(trackPos2, kPiPlus);
307 
308  if (proton1 == 0 || kaonMinus == 0 || pionPlus2 == 0) {
309  pidLcpKpi = 0; //exclude LcpKpi
310  }
311  if (proton1 == 1 && kaonMinus == 1 && pionPlus2 == 1) {
312  pidLcpKpi = 1; //accept LcpKpi
313  }
314  if (proton2 == 0 || kaonMinus == 0 || pionPlus1 == 0) {
315  pidLcpiKp = 0; //exclude LcpiKp
316  }
317  if (proton2 == 1 && kaonMinus == 1 && pionPlus1 == 1) {
318  pidLcpiKp = 1; //accept LcpiKp
319  }
320  }
321 
322  if (pidLcpKpi == 0 && pidLcpiKp == 0) {
323  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
324  continue;
325  }
326 
327  if ((pidLcpKpi == -1 || pidLcpKpi == 1) && topolLcpKpi) {
328  statusLcpKpi = 1; //identified as LcpKpi
329  }
330  if ((pidLcpiKp == -1 || pidLcpiKp == 1) && topolLcpiKp) {
331  statusLcpiKp = 1; //identified as LcpiKp
332  }
333 
334  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
335  }
336  }
