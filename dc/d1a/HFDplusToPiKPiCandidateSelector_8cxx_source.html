69  template <typename T1, typename T2>
70  bool selection(const T1& hfCandProng3, const T2& trackPion1, const T2& trackKaon, const T2& trackPion2)
71  {
72  auto candpT = hfCandProng3.pt();
73  int pTBin = findBin(pTBins, candpT);
74  if (pTBin == -1) {
75  return false;
76  }
77  if (candpT < d_pTCandMin || candpT > d_pTCandMax) {
78  return false; //check that the candidate pT is within the analysis range
79  }
80  if (trackPion1.pt() < cuts->get(pTBin, "pT Pi") || trackKaon.pt() < cuts->get(pTBin, "pT K") || trackPion2.pt() < cuts->get(pTBin, "pT Pi")) {
81  return false; // cut on daughter pT
82  }
83  if (TMath::Abs(InvMassDPlus(hfCandProng3) - RecoDecay::getMassPDG(pdg::code::kDPlus)) > cuts->get(pTBin, "deltaM")) {
84  return false; // invariant mass cut
85  }
86  if (hfCandProng3.decayLength() < cuts->get(pTBin, "decay length")) {
87  return false;
88  }
89  if (hfCandProng3.decayLengthXYNormalised() < cuts->get(pTBin, "normalized decay length XY")) {
90  return false;
91  }
92  if (hfCandProng3.cpa() < cuts->get(pTBin, "cos pointing angle")) {
93  return false;
94  }
95  if (hfCandProng3.cpaXY() < cuts->get(pTBin, "cos pointing angle XY")) {
96  return false;
97  }
98  if (TMath::Abs(hfCandProng3.maxNormalisedDeltaIP()) > cuts->get(pTBin, "max normalized deltaIP")) {
99  return false;
100  }
101  return true;
102  }

139  template <typename T>
140  bool selectionPIDTPC(const T& track, int nPDG, int nSigmaCut)
141  {
142  double nSigma = 100.0; //arbitarily large value
143  nPDG = TMath::Abs(nPDG);
144  if (nPDG == kPiPlus) {
145  nSigma = track.tpcNSigmaPi();
146  } else if (nPDG == kKPlus) {
147  nSigma = track.tpcNSigmaKa();
148  } else {
149  return false;
150  }
151  return nSigma < nSigmaCut;

180  template <typename T>
181  int selectionPID(const T& track, int nPDG)
182  {
183  int statusTPC = -1; //no PID info
184  int statusTOF = -1; //no PID info
185 
186  if (validTPCPID(track)) {
187  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPC)) {
188  statusTPC = 0; //rejected by PID
189  } else {
190  statusTPC = 1; //positive PID
191  }
192  }
193 
194  if (validTOFPID(track)) {
195  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOF)) {
196  statusTOF = 0; //rejected by PID
197  } else {
198  statusTOF = 1; //positive PID
199  }
200  }
201 
202  //conservative PID strategy
203  if (statusTPC == 1 && statusTOF != 0) {
204  return 1;
205  } else if (statusTPC == 0 || statusTOF == 0) {
206  return 0;
207  } else {
208  return -1;
209  }
210  }

212  void process(aod::HfCandProng3 const& hfCandProng3s, aod::BigTracksPID const&)
213  {
214  for (auto& hfCandProng3 : hfCandProng3s) { //looping over 3 prong candidates
215 
216  auto statusDplusToPiKPi = 0; // final selection flag : 0-rejected 1-accepted
217 
218  if (!(hfCandProng3.hfflag() & 1 << DPlusToPiKPi)) {
219  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
220  continue;
221  }
222 
223  auto trackPos1 = hfCandProng3.index0_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
224  auto trackNeg1 = hfCandProng3.index1_as<aod::BigTracksPID>(); //negative daughter (positive for the antiparticles)
225  auto trackPos2 = hfCandProng3.index2_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
226 
227  // daughter track validity selection
228  if (!daughterSelection(trackPos1) || !daughterSelection(trackNeg1) || !daughterSelection(trackPos2)) {
229  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
230  continue;
231  }
232 
233  // topological selection
234  if (!selection(hfCandProng3, trackPos1, trackNeg1, trackPos2)) {
235  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
236  continue;
237  }
238 
239  // pid selection
240  auto pionPlus1 = selectionPID(trackPos1, kPiPlus);
241  auto kaonMinus = selectionPID(trackNeg1, kKPlus);
242  auto pionPlus2 = selectionPID(trackPos2, kPiPlus);
243 
244  if (pionPlus1 == 0 || kaonMinus == 0 || pionPlus2 == 0) { //exclude Dplus for PID
245  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
246  continue;
247  }
248 
249  statusDplusToPiKPi = 1;
250  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
251  }
252  }
