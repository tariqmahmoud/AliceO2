71 int16_t FT0ChannelDataTimeSlotContainer::getAverageTimeForChannel(std::size_t channelID) const
72 {
73 
74  if (0 == mEntriesPerChannel[channelID]) {
75  return 0;
76  }
77 
78  double avg = 0.;
79  for (int iBin = 0; iBin < NUMBER_OF_HISTOGRAM_BINS; ++iBin) {
80  const auto& v = mHistogram.at(iBin, channelID);
81  avg += v * (iBin - HISTOGRAM_RANGE);
82  }
83  return avg / mEntriesPerChannel[channelID];
84 }
