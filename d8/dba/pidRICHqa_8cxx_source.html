42  template <typename T>
43  void makelogaxis(T h)
44  {
45  const int nbins = h->GetNbinsX();
46  double binp[nbins + 1];
47  double max = h->GetXaxis()->GetBinUpEdge(nbins);
48  double min = h->GetXaxis()->GetBinLowEdge(1);
49  if (min <= 0) {
50  min = 0.00001;
51  }
52  double lmin = TMath::Log10(min);
53  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
54  for (int i = 0; i < nbins; i++) {
55  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
56  }
57  binp[nbins] = max + 1;
58  h->GetXaxis()->Set(nbins, binp);
59  }

95  void process(const aod::Tracks& tracks,
96  const aod::TracksExtra& tracksExtra,
97  const aod::McTrackLabels& labels,
98  const aod::RICHs& tracks_rich,
99  const aod::McParticles& mcParticles)
100  {
101  for (const auto& t : tracks_rich) {
102  const auto track = t.track();
103  const auto trackExtra = tracksExtra.iteratorAt(track.globalIndex());
104  if (trackExtra.length() < minLength.value) {
105  continue;
106  }
107  if (trackExtra.length() > maxLength.value) {
108  continue;
109  }
110  const auto mcParticle = labels.iteratorAt(track.globalIndex()).mcParticle();
111  if (pdgCode.value != 0 && abs(mcParticle.pdgCode()) != pdgCode.value) {
112  continue;
113  }
114  if (useOnlyPhysicsPrimary.value == 1 && !MC::isPhysicalPrimary(mcParticles, mcParticle)) { // Selecting primaries
115  histos.fill(HIST("p/Sec"), track.p());
116  continue;
117  }
118  histos.fill(HIST("p/Prim"), track.p());
119  histos.fill(HIST("p/Unselected"), track.p());
120  histos.fill(HIST("pt/Unselected"), track.pt());
121  histos.fill(HIST("qa/signal"), t.richSignal());
122  histos.fill(HIST("qa/signalerror"), t.richSignalError());
123  histos.fill(HIST("qa/deltaEl"), track.p(), t.richDeltaEl());
124  histos.fill(HIST("qa/deltaMu"), track.p(), t.richDeltaMu());
125  histos.fill(HIST("qa/deltaPi"), track.p(), t.richDeltaPi());
126  histos.fill(HIST("qa/deltaKa"), track.p(), t.richDeltaKa());
127  histos.fill(HIST("qa/deltaPr"), track.p(), t.richDeltaPr());
128  histos.fill(HIST("qa/nsigmaEl"), track.p(), t.richNsigmaEl());
129  histos.fill(HIST("qa/nsigmaMu"), track.p(), t.richNsigmaMu());
130  histos.fill(HIST("qa/nsigmaPi"), track.p(), t.richNsigmaPi());
131  histos.fill(HIST("qa/nsigmaKa"), track.p(), t.richNsigmaKa());
132  histos.fill(HIST("qa/nsigmaPr"), track.p(), t.richNsigmaPr());
133  }
134  }
135 };
