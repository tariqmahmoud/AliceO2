70  template <typename T>
71  bool selectionTopol(const T& hfCandProng2)
72  {
73  auto candpT = hfCandProng2.pt();
74  auto pTBin = findBin(pTBins, candpT);
75  if (pTBin == -1) {
76  return false;
77  }
78 
79  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
80  return false; //check that the candidate pT is within the analysis range
81  }
82  if (hfCandProng2.impactParameterProduct() > cuts->get(pTBin, "d0d0")) {
83  return false; //product of daughter impact parameters
84  }
85  if (hfCandProng2.cpa() < cuts->get(pTBin, "cos pointing angle")) {
86  return false; //cosine of pointing angle
87  }
88  if (hfCandProng2.cpaXY() < cuts->get(pTBin, "cos pointing angle xy")) {
89  return false; //cosine of pointing angle XY
90  }
91  if (hfCandProng2.decayLengthXYNormalised() < cuts->get(pTBin, "normalized decay length XY")) {
92  return false; //normalised decay length in XY plane
93  }
94  // if (hfCandProng2.dca() > cuts[pTBin][1]) return false; //candidate DCA
95  //if (hfCandProng2.chi2PCA() > cuts[pTBin][1]) return false; //candidate DCA
96 
97  //decay exponentail law, with tau = beta*gamma*ctau
98  //decay length > ctau retains (1-1/e)
99  if (TMath::Abs(hfCandProng2.impactParameterNormalised0()) < 0.5 || TMath::Abs(hfCandProng2.impactParameterNormalised1()) < 0.5) {
100  return false;
101  }
102  double decayLengthCut = TMath::Min((hfCandProng2.p() * 0.0066) + 0.01, 0.06);
103  if (hfCandProng2.decayLength() * hfCandProng2.decayLength() < decayLengthCut * decayLengthCut) {
104  return false;
105  }
106  if (hfCandProng2.decayLengthNormalised() * hfCandProng2.decayLengthNormalised() < 1.0) {
107  //return false; //add back when getter fixed
108  }
109  return true;
110  }

118  template <typename T1, typename T2>
119  bool selectionTopolConjugate(const T1& hfCandProng2, const T2& trackPion, const T2& trackKaon)
120  {
121  auto candpT = hfCandProng2.pt();
122  auto pTBin = findBin(pTBins, candpT);
123  if (pTBin == -1) {
124  return false;
125  }
126 
127  if (trackPion.sign() > 0) { //invariant mass cut
128  if (TMath::Abs(InvMassD0(hfCandProng2) - RecoDecay::getMassPDG(pdg::code::kD0)) > cuts->get(pTBin, "m")) {
129  return false;
130  }
131  } else {
132  if (TMath::Abs(InvMassD0bar(hfCandProng2) - RecoDecay::getMassPDG(pdg::code::kD0)) > cuts->get(pTBin, "m")) {
133  return false;
134  }
135  }
136 
137  if (trackPion.pt() < cuts->get(pTBin, "pT Pi") || trackKaon.pt() < cuts->get(pTBin, "pT K")) {
138  return false; //cut on daughter pT
139  }
140  if (TMath::Abs(trackPion.dcaPrim0()) > cuts->get(pTBin, "d0pi") || TMath::Abs(trackKaon.dcaPrim0()) > cuts->get(pTBin, "d0K")) {
141  return false; //cut on daughter dca - need to add secondary vertex constraint here
142  }
143 
144  if (trackPion.sign() > 0) { //cut on cos(theta *)
145  if (TMath::Abs(CosThetaStarD0(hfCandProng2)) > cuts->get(pTBin, "cos theta*")) {
146  return false;
147  }
148  } else {
149  if (TMath::Abs(CosThetaStarD0bar(hfCandProng2)) > cuts->get(pTBin, "cos theta*")) {
150  return false;
151  }
152  }
153 
154  return true;
155  }

231  template <typename T>
232  int selectionPID(const T& track, int nPDG)
233  {
234  int statusTPC = -1;
235  int statusTOF = -1;
236 
237  if (validTPCPID(track)) {
238  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPC)) {
239  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPCCombined)) {
240  statusTPC = 0; //rejected by PID
241  } else {
242  statusTPC = 1; //potential to be acceepted if combined with TOF
243  }
244  } else {
245  statusTPC = 2; //positive PID
246  }
247  } else {
248  statusTPC = -1; //no PID info
249  }
250 
251  if (validTOFPID(track)) {
252  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOF)) {
253  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOFCombined)) {
254  statusTOF = 0; //rejected by PID
255  } else {
256  statusTOF = 1; //potential to be acceepted if combined with TOF
257  }
258  } else {
259  statusTOF = 2; //positive PID
260  }
261  } else {
262  statusTOF = -1; //no PID info
263  }
264 
265  if (statusTPC == 2 || statusTOF == 2) {
266  return 1; //what if we have 2 && 0 ?
267  } else if (statusTPC == 1 && statusTOF == 1) {
268  return 1;
269  } else if (statusTPC == 0 || statusTOF == 0) {
270  return 0;
271  } else {
272  return -1;
273  }

276  void process(aod::HfCandProng2 const& hfCandProng2s, aod::BigTracksPID const&)
277  {
278  for (auto& hfCandProng2 : hfCandProng2s) { //looping over 2 prong candidates
279 
280  // final selection flag : 0-rejected 1-accepted
281  int statusD0 = 0;
282  int statusD0bar = 0;
283 
284  if (!(hfCandProng2.hfflag() & 1 << D0ToPiK)) {
285  hfSelD0Candidate(statusD0, statusD0bar);
286  continue;
287  }
288 
289  auto trackPos = hfCandProng2.index0_as<aod::BigTracksPID>(); //positive daughter
290  if (!daughterSelection(trackPos)) {
291  hfSelD0Candidate(statusD0, statusD0bar);
292  continue;
293  }
294  auto trackNeg = hfCandProng2.index1_as<aod::BigTracksPID>(); //negative daughter
295  if (!daughterSelection(trackNeg)) {
296  hfSelD0Candidate(statusD0, statusD0bar);
297  continue;
298  }
299 
300  //conjugate independent topological selection
301  if (!selectionTopol(hfCandProng2)) {
302  hfSelD0Candidate(statusD0, statusD0bar);
303  continue;
304  }
305 
306  //implement filter bit 4 cut - should be done before this task at the track selection level
307  //need to add special cuts (additional cuts on decay length and d0 norm)
308 
309  //conjugate dependent toplogical selection for D0
310  bool topolD0 = selectionTopolConjugate(hfCandProng2, trackPos, trackNeg);
311  //conjugate dependent toplogical selection for D0bar
312  bool topolD0bar = selectionTopolConjugate(hfCandProng2, trackNeg, trackPos);
313 
314  if (!topolD0 && !topolD0bar) {
315  hfSelD0Candidate(statusD0, statusD0bar);
316  continue;
317  }
318 
319  int pionPlus = selectionPID(trackPos, kPiPlus);
320  int kaonMinus = selectionPID(trackNeg, kKPlus);
321  int pionMinus = selectionPID(trackNeg, kPiPlus);
322  int kaonPlus = selectionPID(trackPos, kKPlus);
323 
324  int pidD0 = -1;
325  int pidD0bar = -1;
326 
327  if (pionPlus == 0 || kaonMinus == 0 || pionMinus == 1 || kaonPlus == 1) {
328  pidD0 = 0; //exclude D0
329  }
330  if (pionPlus == 1 || kaonMinus == 1 || pionMinus == 0 || kaonPlus == 0) {
331  pidD0bar = 0; //exclude D0bar
332  }
333  if (pionPlus == 1 && kaonMinus == 1) {
334  pidD0 = 1; //accept D0
335  }
336  if (pionMinus == 1 && kaonPlus == 1) {
337  pidD0bar = 1; //accept D0bar
338  }
339 
340  if (pidD0 == 0 && pidD0bar == 0) {
341  hfSelD0Candidate(statusD0, statusD0bar);
342  continue;
343  }
344 
345  if ((pidD0 == -1 || pidD0 == 1) && topolD0) {
346  statusD0 = 1; //identified as D0
347  }
348  if ((pidD0bar == -1 || pidD0bar == 1) && topolD0bar) {
349  statusD0bar = 1; //identified as D0bar
350  }
351 
352  hfSelD0Candidate(statusD0, statusD0bar);
353  }
354  }
