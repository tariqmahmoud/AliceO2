98 FIT_CALIBRATOR_TEMPLATES
99 typename FIT_CALIBRATOR_TYPE::Slot& FIT_CALIBRATOR_TYPE::emplaceNewSlot(
100  bool front, TFType tstart, TFType tend)
101 {
102 
103  auto& cont = o2::calibration::TimeSlotCalibration<InputCalibrationInfoType, TimeSlotStorageType>::getSlots();
104  auto& slot = front ? cont.emplace_front(tstart, tend) : cont.emplace_back(tstart, tend);
105  slot.setContainer(std::make_unique<TimeSlotStorageType>(*mCurrentCalibObject, mMinEntries));
106  return slot;

119 FIT_CALIBRATOR_TEMPLATES
120 void FIT_CALIBRATOR_TYPE::_storeCurrentCalibrationObject(const CalibrationObjectType& calibrationObject)
121 {
122 
123  static std::map<std::string, std::string> md;
124 
125  auto clName = o2::utils::MemFileHelper::getClassName(calibrationObject);
126  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
127  // end of validity = -1 means currentTimestamp + 1 year
128  mInfoVector.emplace_back(mCalibrationObjectPath, clName, flName, md, ccdb::getCurrentTimestamp(), -1);
129  mCalibrationObjectVector.emplace_back(calibrationObject);
130 }
